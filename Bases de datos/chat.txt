CREATE TABLE SUCURSAL
(Tfno_su Varchar2 (12) PRIMARY KEY,nm
N_su Number (5),
Provincia char (20),
Direccion Varchar2 (50),
DNI Varchar2(9),
FOREIGN KEY (DNI) References Empleado);

CREATE TABLE CLIENTE 
(DNI Varchar2 (9) PRIMARY KEY,
Nombre Char (20),
Apellido Char (20),
Tfno_cl Varchar2 (12),
Pais Char (56),
Tfno_su Varchar2(12),
dia_nacimiento Number(2) CHECK (dia_nacimiento>0 AND dia_nacimiento<32),
mes_nacimiento Number(2) CHECK (mes_nacimiento>0 AND mes_nacimiento<13),
anyo_nacimiento Number(4),
Foreign Key (Tfno_su) References Sucursal);

CREATE TABLE HABITACION
(Numero Number (3) PRIMARY KEY,
tipo Char (10) CHECK (tipo In(‘Quad’,‘Doble’,’King’)),
n_max_inquilinos Number (1) CHECK (n_max_inquilinos>0 AND n_max_inquilinos<7),
metro_cuadrados Number (3));


CREATE TABLE RESERVAR
(DNI Varchar2(9),
Numero Number(3),
metodo Char(8) CHECK (metodo In(‘Internet’,‘Llamada’)),
dia_entrada Number(2) CHECK (dia_entrada>0 AND dia_entrada<32),
mes_entrada Number(2) CHECK (mes_entrada>0 AND mes_entrada<13),
anyo_entrada Number(4) CHECK (anyo_entrada>2022),
dia_salida Number(2) CHECK (dia_salida>0 AND dia_salida<32),
mes_salida Number(2) CHECK (mes_salida>0 AND mes_salida<13),
anyo_salida Number(4) CHECK (anyo_salida>2022),
pension Varchar2(8) CHECK (pension=’Completa’ or pension=’Media’),
PRIMARY KEY (DNI, Numero),
FOREIGN KEY (DNI) References Cliente,
FOREIGN KEY (Numero) References Habitacion);

CREATE TABLE SH
(Tfno_su Varchar2(12),
numero  Number (3),
empleado_sucursar Number (5),
pension Varchar2(8) CHECK (pension=’Completa’ or pension=’Media’),
dia_entrada Number(2) CHECK (dia_entrada>0 AND dia_entrada<32),
mes_entrada Number(2) CHECK (mes_entrada>0 AND mes_entrada<13),
anyo_entrada Number(4) CHECK (anyo_entrada>2022),
dia_salida Number(2) CHECK (dia_salida>0 AND dia_salida<32),
mes_salida Number(2) CHECK (mes_salida>0 AND mes_salida<13),
anyo_salida Number(4) CHECK (anyo_salida>2022),
Primary Key (Tfno_su, numero),
Foreign Key (Tfno_su) References Sucursal);

CREATE TABLE EMPLEADO
(DNI Varchar2(9) PRIMARY KEY,
nombre Char(20),
apellido Char(20),
tlfno_empleado Varchar2(12),
pais Varchar2 (56),
dia_nacimiento Number(2) CHECK (dia_nacimiento>0 AND dia_nacimiento<32),
mes_nacimiento Number(2) CHECK (mes_nacimiento>0 AND mes_nacimiento<13),
anyo_nacimiento Number(4));

CREATE TABLE DEPARTAMENTO
(codigo_dpto Number(5) PRIMARY KEY,
nombre_dpto Varchar2(30),
n_despacho Number(5),
DNI Varchar2(9),
Foreign key (DNI) References Empleado);

CREATE TABLE CURSO
(codigo_curso Number(5) PRIMARY KEY,
nombre_curso Varchar2(30),
n_horas Number(3));

CREATE TABLE ENTIDADBANCARIA
(Codigo_banco Number(5) PRIMARY KEY,
Nombre Char(20),
Direccion Varchar(50));

CREATE TABLE NOMINA
(codigo_nomina Number(10) PRIMARY KEY,
salario_base Number(7),
salario_extra Number(4),
codigo_banco Number(5),
Foreign key (codigo_banco) References Entidadbancaria);

CREATE TABLE CONTRATO
(codigo_contrato Number (10) PRIMARY KEY,
salario_base Number(7),
horas_semanales Number(3),
tipo Char (25) CHECK (tipo=’Permanente’ or tipo=’Tiempo_parcial’ or tipo=’Trabajo_parcial’ or tipo=’Formacion’ or tipo=’Relevo’ or tipo=’Trabajo_fijo_intermitente’));

CREATE TABLE CURSAR
(codigo_curso Number(5),
DNI Varchar2(9),
dia Number(2) CHECK (dia>0 AND dia<32),
mes Number(2) CHECK (mes>0 AND mes<13),
anyo Number(4),
PRIMARY KEY (codigo_curso, DNI),
FOREIGN KEY (DNI) References Empleado,
FOREIGN KEY (codigo_curso) References Curso);

CREATE TABLE ENSENYAR 
(codigo_curso Number(5),
DNI Varchar2(9),
PRIMARY KEY (codigo_curso, DNI),
FOREIGN KEY (codigo_curso) References Curso,
FOREIGN KEY (DNI) References Empleado);

CREATE TABLE TRABAJAR
(codigo_dpto Number(5),
DNI Varchar2(9),
PRIMARY KEY (codigo_dpto, DNI),
FOREIGN KEY (DNI) References Empleado,
FOREIGN KEY (codigo_dpto) References Departamento);

CREATE TABLE ECN
(DNI Varchar2(9),
codigo_contrato Number (10),
codigo_nomina Number(10),
PRIMARY KEY (codigo_contrato, DNI, codigo_nomina),
FOREIGN KEY (DNI) References Empleado,
FOREIGN KEY (codigo_contrato) References Contrato,
FOREIGN KEY (codigo_nomina) References Nomina);

CREATE TABLE PROVEEDOR
(codigo_proveedor Number(10) PRIMARY KEY,
nombre_proveedor Char(20),
origen Varchar2(50));

CREATE TABLE ACTIVIDAD
(codigo_actividad Number(5) PRIMARY KEY,
duracion Number(4),
nombre_actividad Varchar2(50),
lugar Varchar2(50));

CREATE TABLE PRODUCTO
(codigo_producto Number(5) PRIMARY KEY,
nombre_producto Varchar2(50),
precio Number(7),
codigo_proveedor number(10),
FOREIGN KEY (codigo_proveedor) References Proveedor);

CREATE TABLE MONITOR
(DNI_monitor Varchar2(9) PRIMARY KEY,
nombre Char(20),
apellido Char(20),
tlfno_empleado Varchar(12),
pais Varchar2 (56),
dia_nacimiento Number(2) CHECK (dia_nacimiento>0 AND dia_nacimiento<32),
mes_nacimiento Number(2) CHECK (mes_nacimiento>0 AND mes_nacimiento<13),
anyo_nacimiento Number(4),
formacion Varchar2(50));

CREATE TABLE PROPORCIONAR
(DNI_cliente Varchar2(9),
codigo_producto Number(5),
incluido Char(2) CHECK (incluido in(‘si’, ‘no’)),
PRIMARY KEY (codigo_producto, DNI_cliente),
FOREIGN KEY (codigo_producto) References Producto,
FOREIGN KEY (DNI_cliente) References Cliente);

CREATE TABLE ORGANIZAR
(DNI_cliente Varchar2(9),
codigo_proveedor Number(5),
DNI_monitor Varchar2(9),
PRIMARY KEY (codigo_proveedor , DNI_monitor, DNI_cliente),
FOREIGN KEY (codigo_proveedor ) References Proveedor,
FOREIGN KEY (DNI_monitor) References Monitor,
FOREIGN KEY (DNI_cliente) References Cliente);

CREATE TABLE PARTICIPAR
(DNI_cliente Varchar2(9),
codigo_actividad Number(5),
dia Number(2) CHECK (dia>0 AND dia<32),
mes Number(2) CHECK (mes>0 AND mes<13),
anyo Number(4) CHECK (anyo>2022),
n_participantes Number(2) CHECK(n_participantes>5),
PRIMARY KEY (codigo_actividad, DNI_cliente),
FOREIGN KEY (codigo_actividad) References Actividad,
FOREIGN KEY (DNI_cliente) References Cliente);
